# Execute using:
# tesla_dashcam <Location of video files> --output <Location to store encoded video files> @CROSS.txt

# Skip if encoded file already exist.
# --skip_existing

# Delete processed files when video file is encoded.
# --delete_source

# CROSS Layout
--layout CROSS

# Put clip info bottom left
--halign LEFT --valign BOTTOM

# Overlay video with text: event reason, event date/time, event city, # seconds till event, and date/time
# If event reason or event city are empty then do not overlay the text on them
# --text_overlay_fmt "{% if event_reason != '' %}Reason: {{event_reason}}\n{% endif %}Event: {{event_timestamp|timestamp_format}}\n{% if event_city != '' %}City: {{event_city}}\n{% endif %}Countdown: {{event_timestamp_countdown_rolling}}\nDate/Time: {{local_timestamp_rolling}}"

--text_overlay_fmt >
    {% if event_reason != '' %}
        Reason: {{event_reason}}\n
    {% endif %}
    "\      Event: {{event_timestamp|timestamp_format}}\n"
    {% if event_city != '' %}
    "\      City: {{event_city}}\n"
    {% endif %}
    Countdown: {{event_timestamp_countdown_rolling}}\n
    Date/Time: {{local_timestamp_rolling}}

# Set timestamp format to: Weekday abbreviated, day (without leading zero) Month abbreviated Year at Hour:Minute:SecondAM(or PM)
--timestamp_format "%a, %d %b %Y at %I:%M:%S%p"

# Fast forward through portions where there is no motion detected
--motion_only

# Encode with x265
--encoding x265

# Set start offset to 60 seconds before the event
--start_offset -60

# Make it so that offset is only for Sentry and based on when the sentry event occurred.
--sentry_offset

# Set the movie timestamp to SENTRY event
--set_moviefile_timestamp SENTRY

# Store files in folder based on layout, subfolder year, subfolder month, subfolder event reason and event city, 
# --merge is not provided and no extention hence each event is stored seperately with default filename
--output >
    ./Movies/{{layout}}/{{start_timestamp.strftime('%Y')}}/{{start_timestamp|timestamp_format('%m')}}/
    {{ event_reason if event_reason != '' }}
    {{'-' if event_reason != '' and event_city != '' }}
    {{event_city if event_city != ''}}

# Use a temp folder to store files not being kept
--temp_dir "./temp"

# Set timestamp to be year/month
# --output_timestamp_format "%Y-%m"

# Create map
--title_screen_map
